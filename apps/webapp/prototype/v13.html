Developer Experience

Three API calls to
meeting intelligence

Join a meeting, get the transcript, run it through your AI. That's it.

main.py 
import requests

# 1. Join a meeting
bot = requests.post("https://api.cloud.vexa.ai/bots",
    headers={"X-API-Key": API_KEY},
    json={
        "platform": "google_meet",
        "native_meeting_id": "abc-defg-hij",
        "bot_name": "Vexa Notetaker"
    }
)

# 2. Get the transcript
transcript = requests.get(
    "https://api.cloud.vexa.ai/transcripts/google_meet/abc-defg-hij",
    headers={"X-API-Key": API_KEY}
)

# 3. Run through your AI
summary = openai.chat.completions.create(
    model="gpt-4o",
    messages=[{"role": "user", "content": transcript.text()}]
)
