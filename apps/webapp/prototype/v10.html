Developer Experience

Three API calls to
meeting intelligence

Join a meeting, get the transcript, run it through your AI. That's it.

main.py 
import requests

# 1. Join a meeting
bot = requests.post("https://api.cloud.vexa.ai/bots",
    headers={"X-API-Key": API_KEY},
    json={
        "platform": "google_meet",
        "native_meeting_id": "abc-defg-hij",
        "bot_name": "Vexa Notetaker"
    }
)

# 2. Get the transcript
transcript = requests.get(
    "https://api.cloud.vexa.ai/transcripts/google_meet/abc-defg-hij",
    headers={"X-API-Key": API_KEY}
)

# 3. Run through your AI
summary = openai.chat.completions.create(
    model="gpt-4o",
    messages=[{"role": "user", "content": transcript.text()}]
)
terminal 
# 1. Join a meeting
$ curl -X POST https://api.cloud.vexa.ai/bots \
    -H "X-API-Key: YOUR_API_KEY" \
    -H "Content-Type: application/json" \
    -d '{
      "platform": "google_meet",
      "native_meeting_id": "abc-defg-hij",
      "bot_name": "Vexa Notetaker"
    }'

# 2. Get the transcript
$ curl https://api.cloud.vexa.ai/transcripts/google_meet/abc-defg-hij \
    -H "X-API-Key: YOUR_API_KEY"

# 3. Stop the bot
$ curl -X DELETE https://api.cloud.vexa.ai/bots/google_meet/abc-defg-hij \
    -H "X-API-Key: YOUR_API_KEY"
app.ts 
// 1. Join a meeting
const bot = await fetch("https://api.cloud.vexa.ai/bots", {
  method: "POST",
  headers: {
    "X-API-Key": API_KEY,
    "Content-Type": "application/json"
  },
  body: JSON.stringify({
    platform: "google_meet",
    native_meeting_id: "abc-defg-hij",
    bot_name: "Vexa Notetaker"
  })
});

// 2. Get the transcript
const transcript = await fetch(
  "https://api.cloud.vexa.ai/transcripts/google_meet/abc-defg-hij",
  { headers: { "X-API-Key": API_KEY } }
).then(r => r.json());

// 3. Process with AI
const result = await anthropic.messages.create({
  model: "claude-sonnet-4-20250514",
  messages: [{ role: "user", content: transcript.text }]
});
